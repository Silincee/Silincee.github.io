public int findMin(int[] nums) {
  int left = -1;
  int right = nums.length;

  while (right>0 && nums[0] == nums[right-1]) right--;

  while (left+1!=right){
    int m = (right-left)/2+left;

    if (nums[m]>=nums[0]){
      left = m;
    }else {
      right = m;
    }
  }

  if (right==nums.length) return nums[0];

  return nums[right];
}
